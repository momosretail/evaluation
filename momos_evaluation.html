const CLAUDE_API_KEY = 'YOUR_API_KEY_HERE';

function doPost(e) {
  var data = JSON.parse(e.postData.contents);
  if (data.action === 'ai_report') return handleAiReport(data);
  return handleEvalSave(data);
}

function handleEvalSave(data) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(data.sessionName) || ss.insertSheet(data.sessionName);
  if (sheet.getLastRow()===0) sheet.appendRow(['제출시간','평가자','제품명','제품유형',
    '기술의견','기술점수','기술판정','브랜드의견','브랜드점수','브랜드판정',
    '경험의견','경험점수','경험판정','취향의견','취향점수',
    '맛','향','질감','당도','산미',
    '최종결정','보류반대사유','개선우선순위','개선필요']);
  sheet.appendRow([new Date(), data.evaluator, data.productName, data.productType,
    data.c1Comment, data.c1_score, data.c1Verdict,
    data.c2Comment, data.c2_score, data.c2Verdict,
    data.c3Comment, data.c3_score, data.c3Verdict,
    data.c4Comment, data.c4_score,
    data.ds_taste, data.ds_aroma, data.ds_texture, data.ds_sweet, data.ds_acid,
    data.finalVerdict, data.holdReason,
    (data.improvePriority||[]).join(','), data.improvement]);
  return ok();
}

function handleAiReport(data) {
  var res = UrlFetchApp.fetch('https://api.anthropic.com/v1/messages', {
    method:'post', headers:{'Content-Type':'application/json','x-api-key':CLAUDE_API_KEY,'anthropic-version':'2023-06-01'},
    payload: JSON.stringify({model:'claude-sonnet-4-20250514',max_tokens:1000,messages:[{role:'user',content:data.prompt}]})
  });
  var result = JSON.parse(res.getContentText());
  return ContentService.createTextOutput(JSON.stringify({text: result.content[0].text}))
    .setMimeType(ContentService.MimeType.JSON);
}

function ok() {
  return ContentService.createTextOutput(JSON.stringify({result:'ok'}))
    .setMimeType(ContentService.MimeType.JSON);
}
